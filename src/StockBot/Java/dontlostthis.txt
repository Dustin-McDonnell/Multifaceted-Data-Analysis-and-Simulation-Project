 public Double calculateRSI(ArrayList<ArrayList<String>> data){
        Integer n = null;
        Scanner scanner = new Scanner(System.in);
        if (scanner.hasNextInt()){
            n = scanner.nextInt();
        }
        else{
            System.out.println("Enter a number");
        }

        Double avgU = 0.0;
        Double uCounter = 0.0;
        Double avgD = 0.0;
        Double dCounter = 0.0;

        for (int i = data.size() - 1; i >= data.size() - n; i--){
            Double currentClose = Double.parseDouble(data.get(i).get(4));
            System.out.println("Current Close: " + currentClose);
            Double previousClose = Double.parseDouble(data.get(i-1).get(4));
            System.out.println("Previous Close: " + previousClose);
            Double change = currentClose - previousClose;
            if(change > 0){
                avgU = avgU + change;
                System.out.println("UP");
            }
            else{
                avgD = avgD + Math.abs(change);
                System.out.println("Down");
            }
        }
        System.out.println(avgU);
        System.out.println(uCounter);
        System.out.println(avgD);
        System.out.println(dCounter);
        avgU = avgU / n;
        avgD = avgD / n;
        Double relativeStrength = avgU/avgD;
        Double relativeStrengthIndex = 100 - (100/(1+relativeStrength));
        return relativeStrengthIndex;
    }
